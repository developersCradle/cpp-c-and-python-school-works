#include <stdio.h>

int lueIlmentymat();
void tulostaTiedot();



int main(void)
{
	
	
	if(lueIlmentymat() == -1) // Ongelmia tiedoston lukemisessa
	{
		return 0;
	}
	else if(lueIlmentymat() < 50 ) 	//Korkeintaan 50 henkilöä
	{
		tulostaTiedot();
	}

	
	
	return 0;
}

int lueIlmentymat()
{
	int luettu_luku = 0;
	FILE * tied_kahv;
	
	if((tied_kahv = fopen("luettelo.txt", "r")) == NULL)
	{
		printf("Tiedostoa ei löydetty!");
		return -1;
	}
	else 
	{
		fscanf(tied_kahv, "%i", &luettu_luku); 
		fclose(tied_kahv);  
		return luettu_luku;
	}
}

void tulostaTiedot()
{
	char etunimi[21];
	char sukunimi[21];
	char puhelinnumero[21];
	
	int kerrat;
	FILE * tiedosto_kahva;
	
	if((tiedosto_kahva = fopen("luettelo.txt", "r")) == NULL)
	{
		printf("Tiedostoa ei löydetty!");
	}
	else
	{
		
		fscanf(tiedosto_kahva, "%i", &kerrat);
		
		for(int i = 0; i < kerrat; i++)
		{
			fscanf(tiedosto_kahva, "%s %s %s\n", &etunimi, &sukunimi, &puhelinnumero);
			printf("%s %s %s\n", etunimi, sukunimi, puhelinnumero);
		}
		
		
	}
}
